üé® React-Style Props

Easy integration with event handlers and styling:

import { createSliderWithProps } from '@3nvs/polyfront-slider'; const slider = createSliderWithProps({ mode: 'range', min: 0, max: 100, step: 5, showTooltip: true, onChange: (value) => console.log('Changed:', value), onInput: (value) => console.log('Input:', value), style: { width: '100%', margin: '20px 0' } }); document.body.appendChild(slider);
Range: [25, 75]

üì± Framework Integration Examples

Ready-to-use examples for popular frameworks:

React

useEffect + useRef pattern

function PriceSlider({ onChange }) { const sliderRef = useRef(null); useEffect(() => { const slider = createSliderWithProps({ mode: 'range', min: 0, max: 2000, step: 100, onChange: onChange }); sliderRef.current.appendChild(slider); }, [onChange]); return <div ref={sliderRef} />; }

Vue 3

Composition API

<template> <div ref="container" /> </template> <script setup> import { onMounted, ref } from 'vue'; const container = ref(); onMounted(() => { const slider = createSliderWithProps({ mode: 'range', onChange: (value) => emit('change', value) }); container.value.appendChild(slider); }); </script>

Angular

Component lifecycle

@Component({ template: '<div #container></div>' }) export class SliderComponent { @ViewChild('container') container: ElementRef; ngOnInit() { const slider = createSliderWithProps({ mode: 'range', onChange: (value) => this.onChange(value) }); this.container.nativeElement.appendChild(slider); } }
